    private void ShowColoredItemsInDataGridView1()
    {
        // DataGridView의 각 행을 확인하며 주황색과 빨간색인 항목 찾기
        foreach (DataGridViewRow row in DataGridView1.Rows)
        {
            if (row.DefaultCellStyle.ForeColor == Color.Orange || row.DefaultCellStyle.ForeColor == Color.Red)
            {
                // DataGridView의 해당 행과 셀 값을 변경
                foreach (DataGridViewCell cell in row.Cells)
                {
                    cell.Style.ForeColor = Color.Black; // 항목들을 검은색으로 변경
                }
            }
        }
    }
}
